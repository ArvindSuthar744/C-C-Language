--Create City Table 2

CREATE TABLE city (
    id INT PRIMARY KEY,
    city_name VARCHAR(255),
    lat DECIMAL(10, 6),
    long DECIMAL(10, 6),
    country_id INT,
    FOREIGN KEY (country_id) REFERENCES country(id)

--Task : 1 (join multiple tables using left join)

SELECT 
    co.id AS country_id,
    co.country_name_eng AS country_name,
    ci.city_name,
    cu.customer_name
FROM 
    country co
LEFT JOIN 
    city ci ON co.id = ci.country_id
LEFT JOIN 
    customer cu ON ci.id = cu.city_id;



--Task : 2 (join multiple tables using both left and inner join)

SELECT 
    co.id AS country_id,
    co.country_name_eng AS country_name,
    ci.city_name,
    cu.customer_name
FROM 
    country co
LEFT JOIN 
    city ci ON co.id = ci.country_id
LEFT JOIN 
    customer cu ON ci.id = cu.city_id
WHERE 
    ci.id IS NOT NULL
    AND cu.id IS NOT NULL;
